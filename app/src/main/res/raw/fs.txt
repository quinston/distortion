#version 310 es
#extension GL_OES_EGL_image_external : require

precision mediump float;

in vec3 Colour;
in vec2 TextureCoords;
out vec4 outColor;

uniform samplerExternalOES cameraTex;
uniform mat4 textureTransformMatrix;

float clip(float f) {
    if (f > 1.0f) {
        return 1.0f;
    }
    else if (f < 0.0f) {
        return 0.0f;
    }
    else {
        return f;
    }
}

void main() {
    vec4 physicalHomogeneousCoords =  vec4(TextureCoords.x, TextureCoords.y, 0, 1);

    vec4 yuvColour =  texture(cameraTex, physicalHomogeneousCoords.xy);

    outColor = yuvColour;
}
